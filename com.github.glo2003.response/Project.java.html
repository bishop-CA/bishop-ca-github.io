package com.github.glo2003.response;

import com.fasterxml.jackson.annotation.JsonProperty;

import java.util.List;
import java.util.Map;

public class Project {
    @JsonProperty("id")
    public String ID;
    @JsonProperty("name")
    public String name;
    @JsonProperty("lastCommit")
    public String lastCommit;
    @JsonProperty("html_url")
    public String htmlUrl;
    @JsonProperty("ciStatus")
    public String ciStatus;
    @JsonProperty("contributors")
    public List<Contributor> contributors;
    @JsonProperty("languages")
    public Map<String, Integer> languages;
    @JsonProperty("branches")
    public List<Branch> branches;
    @JsonProperty("openIssues")
    public List<Issue> openIssues;
    @JsonProperty("open_issues_count")
    public Integer openIssuesCount;
    @JsonProperty("badges")
    public List<String> badges;


    public Project(final String id, final String name, final List<Contributor> contributors, final Map<String, Integer> languages, final List<Branch> branches, final List<Issue> issues, final int openIssuesCount, final String htmlUrl, final List<String> badges) {
        this.ID = id;
        this.name = name;
        this.contributors = contributors;
        this.languages = languages;
        this.branches = branches;
        this.openIssues = issues;
        this.openIssuesCount = openIssuesCount;
        this.htmlUrl = htmlUrl;
        this.lastCommit = "";
        this.ciStatus = "";
        this.badges = badges;
    }
}
