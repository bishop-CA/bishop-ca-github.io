package com.github.glo2003.response;


import com.fasterxml.jackson.annotation.JsonProperty;

import java.util.HashMap;
import java.util.Map;

public class Branch {
    @JsonProperty("name")
    public String name;
    @JsonProperty("commit")
    public Map<String, String> commit = new HashMap<>();

    public Branch(final String name, final String sha, final String url) {
        this.name = name;
        this.commit.put("sha", sha);
        this.commit.put("url", url);
    }
}
