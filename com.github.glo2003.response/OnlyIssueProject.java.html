package com.github.glo2003.response;

import com.fasterxml.jackson.annotation.JsonProperty;

import java.util.List;

public class OnlyIssueProject {
    @JsonProperty("id")
    public String ID;
    @JsonProperty("name")
    public String name;
    @JsonProperty("html_url")
    public String htmlUrl;
    @JsonProperty("openIssues")
    public List<Issue> openIssues;
    @JsonProperty("open_issues_count")
    public Integer openIssuesCount;


    public OnlyIssueProject(final String id, final String name, final List<Issue> issues, final int openIssuesCount, final String htmlUrl) {
        this.ID = id;
        this.name = name;
        this.openIssues = issues;
        this.openIssuesCount = openIssuesCount;
        this.htmlUrl = htmlUrl;
    }
}
