package com.github.glo2003.response;

import com.fasterxml.jackson.annotation.JsonProperty;

public class Issue {
    @JsonProperty("id")
    public int id;
    @JsonProperty("html_url")
    public String html_url;
    @JsonProperty("title")
    public String title;
    @JsonProperty("number")
    public int number;
    @JsonProperty("state")
    public String state;
    @JsonProperty("body")
    public String body;

    public Issue(final int id, final String htmlUrl, final String title, final int number, final String state, final String body) {
        this.id = id;
        this.html_url = htmlUrl;
        this.title = title;
        this.number = number;
        this.state = state;
        this.body = body;
    }
}
